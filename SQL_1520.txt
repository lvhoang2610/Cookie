SELECT																																									
			t_in_out_whse_schedule.sales_order_no,																																						
			t_in_out_whse_schedule.limit_mng_no ,																																				
			t_in_out_whse_schedule.sales_order_dtl_no,																																						
			t_in_out_whse_schedule.glb_parts_no ,																																			
			t_in_out_whse_schedule.customer_biz_partner_parts_no,																																						
			t_in_out_whse_schedule.mnfct_parts_model ,																																			
			t_in_out_whse_schedule.parts_type ,																																	
			t_sales_order_dtl.order_unit ,																																
			t_sales_order_dtl.order_quantity ,																																	
			t_sales_order_dtl.order_quantity_unit_cd ,																																			
			t_sales_order_dtl.tax_rate ,																															
			t_sales_order_dtl.tax_type ,																															
			m_glb_biz_partner.glb_biz_partner_cd 
FROM																																									
			t_in_out_whse_schedule
INNER JOIN m_glb_biz_partner ON
	m_glb_biz_partner.glb_biz_partner_cd = t_in_out_whse_schedule.sales_glb_biz_partner_cd
	AND m_glb_biz_partner.is_deleted = 0
INNER JOIN t_psi_crt_header ON
	t_psi_crt_header.psi_crt_no = t_in_out_whse_schedule.psi_crt_no
	AND t_psi_crt_header.is_deleted = 0
LEFT OUTER JOIN t_sales_order_dtl ON																													
			t_sales_order_dtl.sales_order_no = t_in_out_whse_schedule.sales_order_no
	AND t_sales_order_dtl.sales_order_dtl_no = t_in_out_whse_schedule.sales_order_dtl_no
	AND t_sales_order_dtl.is_deleted = 0
WHERE																																									
			t_in_out_whse_schedule.auto_id = $autoID
	AND t_in_out_whse_schedule.is_deleted = 0
