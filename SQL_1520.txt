SELECT																																									
			m_customer_credit_limit.credit_limit_amount ,																																				
			m_customer_credit_limit.credit_balance ,																																		
			t_recv_order_balance_ttl.recv_order_balance
FROM																																									
			t_in_out_whse_schedule
LEFT OUTER JOIN m_customer_credit_limit ON																											
			m_customer_credit_limit.ttcg_company_cd = t_in_out_whse_schedule.own_ttcg_company_cd
	AND m_customer_credit_limit.ttcg_section_cd = t_in_out_whse_schedule.in_out_whse_func_section_cd
	AND m_customer_credit_limit.glb_biz_partner_cd = t_in_out_whse_schedule.in_whse_base_glb_biz_partner_cd
	AND m_customer_credit_limit.limit_mng_no = t_in_out_whse_schedule.limit_mng_no
	AND m_customer_credit_limit.currency_cd = t_in_out_whse_schedule.sales_currency_cd
	AND m_customer_credit_limit.is_deleted = 0
LEFT OUTER JOIN t_recv_order_balance_ttl ON																										
			t_recv_order_balance_ttl.ttcg_company_cd = t_in_out_whse_schedule.own_ttcg_company_cd
	AND t_recv_order_balance_ttl.ttcg_section_cd = t_in_out_whse_schedule.in_out_whse_func_section_cd
	AND t_recv_order_balance_ttl.glb_biz_partner_cd = t_in_out_whse_schedule.in_whse_base_glb_biz_partner_cd
	AND t_recv_order_balance_ttl.currency_cd = t_in_out_whse_schedule.sales_currency_cd
	AND t_recv_order_balance_ttl.is_deleted = 0
WHERE																																									
			t_in_out_whse_schedule.is_deleted = 0
	AND t_in_out_whse_schedule.auto_id = '100001'
